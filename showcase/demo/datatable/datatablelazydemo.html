DataTable - Lazy Lazy mode is handy to deal with large datasets, instead of loading the entire data, small chunks of data is loaded by invoking onLazyLoad callback everytime paging, sorting and filtering happens. Sample belows imitates lazy paging by using an in memory list.
List of Cars
 

export class DataTableLazyDemo implements OnInit {

    datasource: Car[];
    
    cars: Car[];

    cols: Column[];
    
    totalRecords: number;

    constructor(private carService: CarService) { }

    ngOnInit() {
        //datasource imitation
        this.carService.getCarsLarge().then(cars => {this.datasource = cars; this.totalRecords = this.datasource.length;});

        this.cols = [
            {field: 'vin', header: 'Vin'},
            {field: 'brand', header: 'Brand'},
            {field: 'year', header: 'Year'},
            {field: 'color', header: 'Color'}
        ];
    }
    
    loadCarsLazy(event) {
        //in a real application, make a remote request to load data using state metadata from event
        this.cars = this.datasource.slice(event.first, (event.first + event.rows));
    }
}



<p-dataTable [value]="cars" [columns]="cols" [lazy]="true" [rows]="10" [paginator]="true" 
                    [totalRecords]="totalRecords" (onLazyLoad)="loadCarsLazy($event)">
    <header>List of Cars</header>
</p-dataTable>
